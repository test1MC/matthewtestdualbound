Status Example

A Status message is used to relay acceptance of a message back to the sender. The following is an example of a Status message.

Status (XML)

<?xml version="1.0" encoding="UTF-8"?>

<transport:Message xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:pa-structures="http://www.ncpdp.org/schema/pa-structures" xmlns:structures="http://www.ncpdp.org/schema/structures" xmlns:script="http://www.ncpdp.org/schema/script" xmlns:datatypes="http://www.ncpdp.org/schema/datatypes" xmlns:transport="http://www.ncpdp.org/schema/transport" xsi:schemaLocation="http://www.ncpdp.org/schema/transport transport.xsd" TransportVersion="v2013111" TransactionVersion="v2013111" PA-StructuresVersion="v2013111" TransactionDomain="SCRIPT" DatatypesVersion="v2013111" ECLVersion="v2013111" StructuresVersion="v2013111">

<transport:Header>

    <transport:To Qualifier="D">1234567890123</transport:To>

    <transport:From Qualifier="PY">1234563</transport:From>

    <transport:MessageID>888333</transport:MessageID>

    <transport:RelatesToMessageID>123</transport:RelatesToMessageID>

    <transport:SentTime>2014-05-16T14:17:12.1Z </transport:SentTime>

    <transport:Security>

      <transport:Sender>

        <transport:TertiaryIdentification>12345</transport:TertiaryIdentification>

      </transport:Sender>

      <transport:Receiver>

        <transport:TertiaryIdentification>678</transport:TertiaryIdentification>

      </transport:Receiver>

    </transport:Security>

    <transport:SenderSoftware>

      <transport:SenderSoftwareDeveloper>MDLITE</transport:SenderSoftwareDeveloper>

      <transport:SenderSoftwareProduct>443</transport:SenderSoftwareProduct>

      <transport:SenderSoftwareVersionRelease>2.1</transport:SenderSoftwareVersionRelease>

    </transport:SenderSoftware>

  </transport:Header>

<transport:Body>

   <transport:Status>

      <transport:Code>010</transport:Code>

    </transport:Status>

  </transport:Body>

</transport:Message>
