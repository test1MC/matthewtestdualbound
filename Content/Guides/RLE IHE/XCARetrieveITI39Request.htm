The following is an example of a XCA Retrieve Request Message sent from the requesting location to Surescripts.

<s:Envelope xmlns:s="http://www.w3.org/2003/05/soap-envelope" xmlns:a="http://www.w3.org/2005/08/addressing" xmlns:u="http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-utility-1.0.xsd" xmlns:urn="urn:ihe:iti:xds-b:2007">

   <s:Header>

      <a:Action s:mustUnderstand="1">urn:ihe:iti:2007:CrossGatewayQuery</a:Action>

      <a:MessageID>urn:uuid:ew8d2w81-95we-h2w8-71e5-25w5f1sd6f1</a:MessageID>

      <a:ReplyTo>

         <a:Address>http://www.w3.org/2005/08/addressing/anonymous</a:Address>

      </a:ReplyTo>

      <a:To s:mustUnderstand="1" u:Id="_1">https://test.abc.com/abc.community/xca.svc</a:To>

      <a:From a:IsReferenceParameter="true">

         <a:Address>urn:abc:123.xcarequest</a:Address>

      </a:From>

      <o:Security s:mustUnderstand="1" xmlns:o="http://docs.oasis-open.org/wss/2004/01/oasis-200401-wss-wssecurity-secext-1.0.xsd">

         <u:Timestamp u:Id="_0">

            <u:Created>2013-11-05T19:38:58.953Z</u:Created>

            <u:Expires>2013-11-05T19:43:58.953Z</u:Expires>

         </u:Timestamp>

         <Assertion ID="_cba9fd90-059c-4963-acbd-24b3ae8b7622" IssueInstant="2013-11-05T19:38:58.937Z" Version="2.0" xmlns="urn:oasis:names:tc:SAML:2.0:assertion">

                       <AttributeStatement>

               <Attribute Name="urn:oasis:names:tc:xspa:1.0:subject:organization">

                  <AttributeValue>Originator Health Clinic</AttributeValue>

               </Attribute>

               <Attribute Name="urn:oasis:names:tc:xspa:1.0:subject:organization-id">

                  <AttributeValue>urn:oid:3.2.132.132132.1.32.13.2.1.3.321321.123</AttributeValue>

               </Attribute>

            </AttributeStatement>

         </Assertion>

      </o:Security>

   </s:Header>

   <s:Body>

      <urn:RetrieveDocumentSetRequest><!--One Document request-->

       <urn:DocumentRequest>

          <urn:HomeCommunityId>urn:oid:2.16.840.1.113883.3.2054.2.1</urn:HomeCommunityId>

       </urn:DocumentRequest>

      </urn:RetrieveDocumentSetRequest>

   </s:Body>

</s:Envelope>
